25 static void launchSoftAp(){
26  esp_netif_t *ap_netif = NULL;
27  esp_netif_init();
28  ESP_ERROR_CHECK(esp_event_loop_create_default());
29  ap_netif = esp_netif_create_default_wifi_ap();
30  ESP_ERROR_CHECK(esp_netif_dhcps_stop(ap_netif));
31  esp_netif_ip_info_t ipAddressInfo;
32  memset(&ipAddressInfo, 0, sizeof(ipAddressInfo));
33  IP4_ADDR(
34  &ipAddressInfo.ip,
35  SOFT_AP_IP_ADDRESS_1,
36  SOFT_AP_IP_ADDRESS_2,
37  SOFT_AP_IP_ADDRESS_3,
38  SOFT_AP_IP_ADDRESS_4);
39  IP4_ADDR(
40  &ipAddressInfo.gw,
41  SOFT_AP_GW_ADDRESS_1,
42  SOFT_AP_GW_ADDRESS_2,
43  SOFT_AP_GW_ADDRESS_3,
44  SOFT_AP_GW_ADDRESS_4);
45  IP4_ADDR(
46  &ipAddressInfo.netmask,
47  SOFT_AP_NM_ADDRESS_1,
48  SOFT_AP_NM_ADDRESS_2,
49  SOFT_AP_NM_ADDRESS_3,
50  SOFT_AP_NM_ADDRESS_4);
51 
52 
53  ESP_ERROR_CHECK(esp_netif_set_ip_info(ap_netif, &ipAddressInfo));
54  ESP_ERROR_CHECK(esp_netif_dhcps_start(ap_netif));
55 // ESP_ERROR_CHECK(esp_event_loop_init(wifiEventHandler, NULL));
56  ESP_ERROR_CHECK(esp_event_handler_instance_register(WIFI_EVENT,
57  ESP_EVENT_ANY_ID,
58  &wifi_event_handler,
59  NULL,
60  NULL));
61  wifi_init_config_t wifiConfiguration = WIFI_INIT_CONFIG_DEFAULT();
62  ESP_ERROR_CHECK(esp_wifi_init(&wifiConfiguration));
63  ESP_ERROR_CHECK(esp_wifi_set_mode(WIFI_MODE_AP));
64  wifi_config_t apConfiguration = {
65  .ap = {
66  .ssid = WIFI_AP_SSID,
67  .password = WIFI_AP_PASSWORD,
68  .ssid_len = strlen(WIFI_AP_SSID),
69  .authmode = WIFI_AUTH_WPA_WPA2_PSK,
70  //.ssid_hidden = 0,
71  .max_connection = WIFI_MAX_STA,
72  .beacon_interval = 150,
73  },
74  };
75 
76  esp_netif_get_ip_info(ap_netif, &ipAddressInfo);
77 
78  // power save disabled
79  esp_wifi_set_ps(WIFI_PS_NONE);
80 
81  printf("IP: %d.%d.%d.%d:%d", ipAddressInfo.ip.addr & 0xFF, (ipAddressInfo.ip.addr >> 8) & 0xFF, (ipAddressInfo.ip.addr >> 16) & 0xFF, (ipAddressInfo.ip.addr >> 24) & 0xFF, CONFIG_ESP_HTTP_SERVER_PORT);
82 
83  ESP_ERROR_CHECK(esp_wifi_set_config(WIFI_IF_AP, &apConfiguration));
84  ESP_ERROR_CHECK(esp_wifi_set_storage(WIFI_STORAGE_RAM));
85  ESP_ERROR_CHECK(esp_wifi_start());
86 }
