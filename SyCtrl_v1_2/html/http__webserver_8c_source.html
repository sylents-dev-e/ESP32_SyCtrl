214  if (strcmp("/get", req->uri) == 0) {
215  httpd_resp_send_err(req, HTTPD_404_NOT_FOUND,
216  "/get URI is not available");
217  /* Return ESP_OK to keep underlying socket open */
218  return ESP_OK;
219  } else if (strcmp("/post", req->uri) == 0) {
220  httpd_resp_send_err(req, HTTPD_404_NOT_FOUND,
221  "/post URI is not available");
222  /* Return ESP_FAIL to close underlying socket */
223  return ESP_FAIL;
224  }
225  /* For any other URI send 404 and close socket */
226  httpd_resp_send_err(req, HTTPD_404_NOT_FOUND, "Some 404 error message");
