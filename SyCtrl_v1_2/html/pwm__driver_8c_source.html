59 void pwm_throttle_task(void *arg) {
60 
61 #if DEBUG_TASK_STACK
62  UBaseType_t uxHighWaterMark;
63  uxHighWaterMark = uxTaskGetStackHighWaterMark( NULL);
64 #endif
65 
66 #if WDT_PWM
67  // WTD init for that task
68  CHECK_ERROR_CODE(esp_task_wdt_add(NULL), ESP_OK);
69  CHECK_ERROR_CODE(esp_task_wdt_status(NULL), ESP_OK);
70 #endif
71 
72  while (1) {
73 
74 #if WDT_PWM
75  // give response to WDT
76  CHECK_ERROR_CODE(esp_task_wdt_reset(), ESP_OK);
77 #endif
78 
79 #if PWM_THROTTLE
80 
81  uint32_t c = 0;
82  int32_t mode_cnt = 0;
83 
84  // receiving sempahore and queue data from button ISR
85  // TODO: maybe better with portMAX_DELAY but then WDT will alert
86  if ((xSemaphoreTake( CountSemaphore, portMAX_DELAY ) == pdTRUE)
87  && (xQueueReceive(queue_button, &c, portMAX_DELAY) == pdPASS)) {
88 
89  if (c == EMERGENCY) {
90  // emergency stop
91  forward_set_duty(MODE1, mode_cnt);
92  // reset count variable
93  mode_cnt = 0;
94  } else {
95  // counter variable to figure out which duty mode should be set
96  if ((c == UP) && (mode_cnt < 4)) {
97  mode_cnt++;
98  c = 0;
99  } else if ((c == DOWN) && (mode_cnt > -1)) {
100  mode_cnt--;
101  c = 0;
102  } else {
103  ;
104  }
105 
106  /* setting pwm-cycle depending on counter variable *mode_cnt*
107  * switch case for faster response on several cases instead of
108  * if/else if
109  */
110  switch (mode_cnt) {
111  case -1:
112 #if DEBUG_PRINT_BUTTON
113  printf("count: %d\t not implemented\n", mode_cnt);
114  break; //not yet implemented
115 #endif
116  case 0:
117  forward_set_duty(MODE1, mode_cnt);
118  break;
119  case 1:
120  forward_set_duty(PWM_START_CURRENT, mode_cnt);
121  break;
122  case 2:
123  forward_set_duty((PWM_INC * mode_cnt) + PWM_START_CURRENT,
124  mode_cnt);
125  break;
126  case 3:
127  forward_set_duty((PWM_INC * mode_cnt) + PWM_START_CURRENT,
128  mode_cnt);
129  break;
130  case 4:
131  forward_set_duty((PWM_INC * mode_cnt) + PWM_START_CURRENT,
132  mode_cnt);
133  break;
134  default:
135  forward_set_duty((PWM_INC * mode_cnt) + PWM_START_CURRENT,
136  mode_cnt);
137  break;
138  }
139  }
140 
141  // send gear data to display
142  xQueueSend(queue_to_ili9341, &mode_cnt, (TickType_t ) 10);
143 
144  // Send data to http server --> JSON to App
145  xQueueSend(queue_to_json_to_app, &mode_cnt, (TickType_t ) 10);
146 
147  /* Wait a short time then clear any pending button pushes as a crude
148  method of debouncing the switch. xSemaphoreTake() uses a block time of
149  zero this time so it returns immediately rather than waiting for the
150  interrupt to occur. */
151 
152  vTaskDelay(100/ portTICK_PERIOD_MS);
153 
154  xSemaphoreTake(CountSemaphore, 0);
155 
156  }
157 #endif
158 
159 #if DEBUG_TASK_STACK
160  uxHighWaterMark = uxTaskGetStackHighWaterMark( NULL);
161  ets_printf("%d KB %s\n", uxHighWaterMark, "pwm_throtte_task");
162 #endif
163  }
164 
165  vTaskDelete(NULL);
166 }
